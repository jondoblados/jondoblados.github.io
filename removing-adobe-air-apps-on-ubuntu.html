jon@jedi-ntbk:/opt$ sudo dpkg -S /opt/*adobeair-enu: /opt/Adobe AIRcom.agileagenda.agileagenda.f49a4d8df78a1fee7a3be440dc11bab18d922274.1: /opt/AgileAgendade.makesoft.twhirl.0ea062bc275e7ed1e6ec3762effd73c7158adf33.1: /opt/twhirl
